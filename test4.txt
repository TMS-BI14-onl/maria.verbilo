--5
SELECT TOP 5 WITH TIES name
FROM 
(
SELECT u.name, user_id, COUNT(product_id) OVER (PARTITION BY user_id) AS cnt
FROM [user] u JOIN [order] o ON u.id=o.user_id  
)
ORDER BY cnt
