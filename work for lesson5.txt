--Задача 1

SELECT [ФИО водителя], [Стаж]
FROM Водители
WHERE [ФИО водителя] NOT IN ('3','5','6','10') 
		     AND [Стаж]='еее'

--Задача 2
SELECT [№ паспорта]
FROM [Диспетчерская] AS d
LEFT JOIN [Водители] AS v ON d.[Код водителя]=v.[Код водителя]
WHERE [Путевка] IS NULL

--Задача 3
SELECT [Регистрационный № авто], [Время прибытия], [Время отбытия], [Пункт назначения]
FROM [Aвто] AS a
JOIN [Диспеческая] AS d ON a.[Код авто]=d.[Код авто]
JOIN [Маршрут] AS m ON a.[Путевка]=m.[Код маршрута]
WHERE [Год выпуска]='2015'
